class AppPermissionEngine(PermissionEngine):
    def init(self, roles: Dict[str, List[str]] | None = None, active_role: str = "system"):
        self._roles = roles or {
            "system": [
                "system.read",
                "system.write",
            ],
        }
        self._active_role = active_role

    def set_role(self, role: str) -> None:
        self._active_role = role

    def check_permission(self, permission_level: str, context: Dict[str, Any] | None = None) -> bool:
        allowed = self._roles.get(self._active_role, [])
        return permission_level in allowed


class AppCommand:
    def init(self, name: str, permission_level: str, handler):
        self.name = name
        self.permission_level = permission_level
        self._handler = handler

    def handler(self, params: Dict[str, Any]) -> Any:
        return self._handler(params)


class AppCommandRegistry(CommandRegistry):
    def init(self):
        self._commands: Dict[str, AppCommand] = {}

    def register(self, cmd: AppCommand) -> None:
        self._commands[cmd.name] = cmd

    def get_command(self, name: str) -> AppCommand | None:
        return self._commands.get(name)
