import AppIntents
import Foundation

// MARK: - Configuration
private let carinaBridgeURL = "http://localhost:5050/execute"  // ← update for prod

// MARK: - App Intent

struct RunCarinaCommand: AppIntent {

    static var title: LocalizedStringResource = "Run Carina Command"
    static var description = IntentDescription("Send a command to the Carina agent via the local bridge.")

    @Parameter(title: "Command", description: "The command name to execute (e.g. 'status', 'flush_cache')")
    var command: String

    @Parameter(title: "Arguments", description: "Optional arguments as a comma-separated string", default: "")
    var arguments: String

    @Parameter(title: "Role", description: "Caller role: guest | operator | admin | root", default: "operator")
    var role: String

    @Parameter(title: "Force Execute", description: "Bypass GUARDED/RESTRICTED permission gates", default: false)
    var forceExecute: Bool

    @Parameter(title: "Reason", description: "Required for RESTRICTED commands", default: "")
    var reason: String

    static var parameterSummary: some ParameterSummary {
        Summary("Run Carina command \(\.$command) as \(\.$role)")
    }

    func perform() async throws -> some ReturnsValue<String> & ProvidesDialog {
        let payload: [String: Any] = [
            "command":       command,
            "arguments":     arguments,
            "role":          role,
            "force_execute": forceExecute,
            "reason":        reason
        ]

        guard let url = URL(string: carinaBridgeURL) else {
            throw CarinaError.invalidURL
        }

        var request = URLRequest(url: url, timeoutInterval: 10)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = try JSONSerialization.data(withJSONObject: payload)

        let (data, response) = try await URLSession.shared.data(for: request)

        guard let http = response as? HTTPURLResponse else {
            throw CarinaError.invalidResponse
        }

        let body = try JSONDecoder().decode(BridgeResponse.self, from: data)

        if http.statusCode == 200 {
            return .result(value: body.result, dialog: "\(body.result)")
        } else {
            return .result(value: body.result, dialog: "Carina denied: \(body.result)")
        }
    }
}

// MARK: - Models

struct BridgeResponse: Codable {
    let result: String
    let command: String
    let allowed: Bool
}

// MARK: - Errors

enum CarinaError: Error, LocalizedError {
    case invalidURL
    case invalidResponse

    var errorDescription: String? {
        switch self {
        case .invalidURL:       return "Carina bridge URL is invalid."
        case .invalidResponse:  return "Unexpected response from Carina bridge."
        }
    }
}

def build_registry() -> PermissionEngine:
    """
    Instantiate and populate the permission engine with all registered commands.
    Returns a ready-to-use PermissionEngine.
    """
    engine = PermissionEngine()

    commands = [
        Command("ping",           PermissionLevel.OPEN,       _ping,           "Health check"),
        Command("status",         PermissionLevel.OPEN,       _status,         "System status snapshot"),
        Command("list_commands",  PermissionLevel.OPEN,       _list_commands,  "List all registered commands"),
        Command("read_config",    PermissionLevel.GUARDED,    _read_config,    "Read current system config"),
        Command("flush_cache",    PermissionLevel.RESTRICTED, _flush_cache,    "Flush all caches"),
        Command("restart_agent",  PermissionLevel.RESTRICTED, _restart_agent,  "Restart the Carina agent process"),
        Command("kill_process",   PermissionLevel.LOCKED,     _kill_process,   "Hard-kill a running process"),
    ]

    for cmd in commands:
        engine.register(cmd)

    return engine

from flask import Flask, request, jsonify, Response
from RTOS_Permission_Engine import CallerContext
from command_registry import build_registry

app = Flask(name)
engine = build_registry()   # Built once at startup — thread-safe reads

made with Sesame

@app.route("/execute", methods=["POST"])
def execute() -> tuple[Response, int]:
    """
    Execute a Carina command.

    Expected JSON body:
    {
        "command":       "flush_cache",
        "role":          "admin",
        "reason":        "nightly maintenance",
        "force_execute": false,
        "arguments":     ""          // optional, reserved for future use
    }
    """
    data = request.get_json(force=True, silent=True)
    if not data:
        return jsonify({"error": "Invalid or missing JSON body"}), 400

    command       = data.get("command", "").strip()
    role          = data.get("role", "guest").strip()
    reason        = data.get("reason", "").strip()
    force_execute = bool(data.get("force_execute", False))

    if not command:
        return jsonify({"error": "Missing required field: 'command'"}), 400

    caller = CallerContext(
        role=role,
        force_execute=force_execute,
        reason=reason,
    )

    result  = engine.execute(command, caller)
    allowed = not result.startswith("[DENY]")

    status_code = 200 if allowed else 403
    return jsonify({
        "command": command,
        "result":  result,
        "allowed": allowed,
    }), status_code


@app.route("/health", methods=["GET"])
def health() -> tuple[Response, int]:
    """Liveness probe — used by iOS Swift client and monitoring."""
    return jsonify({"status": "ok", "service": "carina-bridge"}), 200


@app.route("/commands", methods=["GET"])
def list_commands() -> tuple[Response, int]:
    """Return all registered command names (no handlers or permission details)."""
    names = list(engine._registry.keys())
    return jsonify({"commands": names}), 200


@app.route("/audit", methods=["GET"])
def audit_log() -> tuple[Response, int]:
    """Return the full in-memory audit log. Swap for DB-backed log in prod."""
    return jsonify({"log": engine.audit_log()}), 200
